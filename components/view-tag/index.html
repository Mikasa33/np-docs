演示 ​

基础 ​

格式化并展示标签。

男 male
male
查看代码
vue
<script lang="ts" setup>
import { NFlex } from 'naive-ui'
import { NpViewTag } from '@mikasa33/np'

const value = 'male'
const options = [
  { label: '男 male', value: 'male' },
  { label: '女 female', value: 'female' },
]
</script>

<template>
  <NFlex>
    <NpViewTag
      :value
      :options
      round
      type="success"
    />
    <NpViewTag :value />
  </NFlex>
</template>
